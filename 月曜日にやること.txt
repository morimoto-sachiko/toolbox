using System;
using System.IO;
using System.Text;
using System.Text.Json;

namespace NormanJsonFormatter
{
    public static class Program
    {
        static void Main(string[] args)
        {
            if (args.Length < 2)
            {
                Console.WriteLine("Usage: NormanJsonFormatter <input.json> <output.json>");
                return;
            }

            string inputPath = args[0];
            string outputPath = args[1];

            string jsonText = File.ReadAllText(inputPath);

            using var document = JsonDocument.Parse(jsonText);

            string result = ConvertToNormanJson(document.RootElement);

            // ★ UTF-8（BOMなし）＋ LF固定
            var utf8NoBom = new UTF8Encoding(encoderShouldEmitUTF8Identifier: false);
            File.WriteAllText(outputPath, result, utf8NoBom);

            Console.WriteLine("Convert Complete.");
        }

        private static string ConvertToNormanJson(JsonElement root)
        {
            var sb = new StringBuilder(1024);
            var writer = new NormanJsonWriter(sb);

            WriteElement(root, writer, isInArray: false, indentLevel: 0);

            return sb.ToString();
        }

        private static void WriteElement(JsonElement element, NormanJsonWriter writer, bool isInArray, int indentLevel)
        {
            switch (element.ValueKind)
            {
                case JsonValueKind.Object:
                    WriteObject(element, writer, isInArray, indentLevel);
                    break;

                case JsonValueKind.Array:
                    writer.Write("[\n");

                    bool first = true;
                    foreach (var item in element.EnumerateArray())
                    {
                        if (!first)
                            writer.Write(",\n");

                        writer.WriteIndent(indentLevel + 1);
                        WriteElement(item, writer, isInArray: true, indentLevel + 1);

                        first = false;
                    }

                    writer.Write("\n");
                    writer.WriteIndent(indentLevel);
                    writer.Write("]");
                    break;

                case JsonValueKind.String:
                    writer.Write(JsonSerializer.Serialize(element.GetString()));
                    break;

                case JsonValueKind.Number:
                case JsonValueKind.True:
                case JsonValueKind.False:
                    writer.Write(element.GetRawText());
                    break;

                case JsonValueKind.Null:
                    writer.Write("null");
                    break;
            }
        }

        private static void WriteObject(JsonElement element, NormanJsonWriter writer, bool isInArray, int indentLevel)
        {
            if (isInArray)
            {
                writer.Write("  "); // ← 仕様対応
            }

            writer.Write("{\n");

            bool first = true;
            foreach (var prop in element.EnumerateObject())
            {
                if (!first)
                    writer.Write(",\n");

                writer.WriteIndent(indentLevel + 1);

                writer.Write(JsonSerializer.Serialize(prop.Name));
                writer.Write(": ");

                WriteElement(prop.Value, writer, isInArray: false, indentLevel + 1);

                first = false;
            }

            writer.Write("\n");
            writer.WriteIndent(indentLevel);
            writer.Write("}");
        }
    }

    public class NormanJsonWriter
    {
        private readonly StringBuilder _sb;
        private const string IndentUnit = "  ";

        public NormanJsonWriter(StringBuilder sb)
        {
            _sb = sb;
        }

        public void Write(string text)
        {
            // LF固定
            _sb.Append(text.Replace("\r\n", "\n").Replace("\r", "\n"));
        }

        public void WriteIndent(int level)
        {
            for (int i = 0; i < level; i++)
            {
                _sb.Append(IndentUnit);
            }
        }
    }
}
