using System;
using System.Collections.Generic;
using System.IO;
using System.Text.Json;
using System.Text.Json.Nodes;

public class PlayerParam
{
    public string Name { get; set; }
    public int Hp { get; set; }
}

public static class JsonReplaceExample
{
    public static void ReplacePlayersArray(string jsonPath, List<PlayerParam> newPlayers)
    {
        // JSON読み込み
        var jsonText = File.ReadAllText(jsonPath);

        JsonNode root = JsonNode.Parse(jsonText);

        if (root == null)
            throw new Exception("JSON parse failed");

        JsonObject obj = root.AsObject();

        // 新しいPlayers配列を作成
        JsonArray newArray = new JsonArray();

        foreach (var p in newPlayers)
        {
            var node = new JsonObject
            {
                ["Name"] = p.Name,
                ["Hp"] = p.Hp
            };

            newArray.Add(node);
        }

        // 配列差し替え
        obj["Players"] = newArray;

        // 保存
        var options = new JsonSerializerOptions
        {
            WriteIndented = true
        };

        var result = obj.ToJsonString(options);

        File.WriteAllText(jsonPath, result);
    }
}
